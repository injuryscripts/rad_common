#initialize(column: nil, options: nil, grouped: false, scope_values: nil, joins: nil, input_label: nil, default_value: nil, blank_value_label: nil, scope: nil, multiple: false) ⇒ SearchFilter

Returns a new instance of SearchFilter.

Examples:

active record list

options: User.all

Two dimensional array options

options: [['Option 1', 1],['Option 2', 2], ['Option 3', 3]]

Grouped options mixed with scoped values

options: [['...', [user, { scope_value: :unassigned }]],
         ['Active', User.active.by_name],
         ['Inactive', User.inactive.by_name]]

Using scope values

[{ column: :owner_id, options: User.by_name, scope_values: { 'Pending Values': :pending } }]

Parameters:

  • column (defaults to: nil) —

    the database column that is being filtered

  • options (defaults to: nil) —

    the options to be displayed in the dropdown. This is required when no scope values are specified.

  • grouped (defaults to: false) —

    will the options be grouped

  • scope_values (defaults to: nil) —

    An array of scopes active record scopes represented by symbols to be inserted into the dropdown options.For example :closed_orders, :open_orders. This is required when no options are specified.

  • joins (defaults to: nil) —

    any necessary sql joins so that the query can be performed

  • input_label (optional) (defaults to: nil) —

    by default the input label for the dropdown is determined by the column name but you can override that by specifying it here

  • default_value (optional) (defaults to: nil) —

    the default value selected in the dropdown options

  • blank_value_label (optional) (defaults to: nil) —

    the text displayed in the dropdown when there is no value selected. By default this is calculated based on the query model name but can be overidden here

  • scope (optional) (defaults to: nil)
  • multiple (Boolean optional) (defaults to: false) —

    when set to true the dropdown becomes a multiple select

def initialize(column: nil, options: nil, grouped: false, scope_values: nil, joins: nil, input_label: nil, default_value: nil, blank_value_label: nil, scope: nil, multiple: false)
  if input_label.blank? && !options.respond_to?(:table_name)
    raise 'Input label is required when options are not active record objects'
  end

  raise 'options or scope_values' if options.nil? && scope_values.nil?

  @column = column
  @options = options
  @joins = joins
  @input_label = input_label
  @blank_value_label = blank_value_label
  @scope_values = scope_values
  @scope = scope
  @multiple = multiple
  @default_value = default_value
  @grouped = grouped
end

def apply_filter(results, search_params)
  value = filter_value(search_params)
  if scope_search?
    apply_scope_filter(results, value)
  elsif scope_value?(value)
    apply_scope_value(results, value)
  elsif value.present?
    if value.is_a? Array
      values = convert_array_values(value)
      results.where("#{searchable_name} IN (?)", values) if values.present?
    else
      results.where("#{query_column(results)} = ?", value)
    end
  end
end

def input_options
  if grouped_scope_values?
    options.map do |option|
      group_key = option.first
      group_options = option.second
      [group_key, transform_group_options(group_options)]
    end
  elsif scope_values?
    scope_options = if @scope_values.is_a? Array
                      @scope_values.map { |option| [option.to_s.titleize, option.to_s] }
                    else
                      @scope_values.keys.map { |option| [option.to_s, option.to_s] }
                    end
    scope_options += options.map { |option| [option.to_s, option.id] } if options.present?
    scope_options
  else
    options
  end
end

def query_column(results)

  "#{results.table_name}.#{searchable_name}"
end

def transform_group_options(group_options)
  group_options.map do |option|
    if scope_value_option?(option)
      [option[:scope_value].to_s.titleize, option[:scope_value].to_s]
    else
      [option.to_s, option.id]
    end
  end
end
